RobotProperties{
RobotName: BumbleBee;
RobotType: AdvancedRobot;
GunColor: yellow;
RadarColor: blue;
BodyColor:yellow;
}

strategy Default(){

    number runs = AmountOfRuns();

    routine(runs){
        setTurnGunRight(50);
        execute();

    }

    when (ScannedRobotEvent e){
        setTurnGunLeft(50);
        setFire(2);
        execute();
    }
}

number AmountOfRuns(){

    text some = RecieveSomeText();
    if(some == "SomeText"){
        return 5;
    }
    else if(some != "SomeText"){
        return 8;
    }

    return 0;

}

text RecieveSomeText(){

    return "SomeText";
}
